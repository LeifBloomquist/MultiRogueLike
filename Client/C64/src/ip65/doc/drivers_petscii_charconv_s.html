;given a PETSCII char in A, return equivalent ASCII

  tax
  lda petscii_to_ascii_table,x
  rts

;given an ASCII char in A, return equivalent PETSCII
ascii_to_native:
  tax
  lda ascii_to_petscii_table,x
  rts
